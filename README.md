## Usage

```javascript
import {safeGet, safeSet, hasOwnProperty} from 'object-extensions'
const obj = {
  a: 1,
  b: [{
    c: 'd', 
    e: ['f']
  }]
}
hasOwnProperty(obj, 'b.0.c') // true
hasOwnProperty(obj, 'a.1.2') // false
safeGet(obj, 'b.0.c') // 'd'
safeGet(obj, 'a.1.2') // undefined
safeSet(obj, 'b.0.c', 1) // true
safeSet(obj, 'a.1.2', 1) // true
```
